                            // 检查用户输入是否正确
                            const userInput = this.parentElement.querySelector('.user-input').value.trim().toLowerCase();
                            const correctAnswer = answerDiv.textContent.trim().toLowerCase();
                            
                            // 移除之前的反馈
                            const oldFeedback = answerDiv.querySelector('.feedback');
                            if (oldFeedback) {
                                oldFeedback.remove();
                            }
                            
                            // 添加新的反馈
                            const feedback = document.createElement('div');
                            feedback.className = 'feedback';
                            
                            if (userInput === '') {
                                feedback.textContent = '您还没有输入翻译';
                            } else if (userInput === correctAnswer) {
                                feedback.className += ' correct';
                                feedback.textContent = '正确！';

                                feedback.textContent = '您的翻译与参考答案不同';

                            answerDiv.appendChild(feedback);

                            updateProgress();
                        }

                        showButton.textContent = '隐藏答案';

                        // 移除之前的反馈
                        const oldFeedback = answerDiv.querySelector('.feedback');
                        if (oldFeedback) {
                            oldFeedback.remove();

                        } else if (userInput === correctAnswer) {
                            feedback.className += ' correct';

                        } else {

                    const inputs = document.querySelectorAll('.user-input');
                    
                    answerDivs.forEach(answerDiv => {
                        answerDiv.style.display = 'none';
                        const showButton = answerDiv.previousElementSibling;
                        showButton.textContent = '查看答案';
                        
                        // 移除反馈
                        const oldFeedback = answerDiv.querySelector('.feedback');
                        if (oldFeedback) {
                            oldFeedback.remove();
                        }
                    });
                    
                    inputs.forEach(input => {
                        input.value = '';
                    });
                    
                    // 重置进度
                    document.getElementById('completed').textContent = '0';
                });
                
                // 更新进度函数
                function updateProgress() {
                    const inputs = document.querySelectorAll('.user-input');
                    let completed = 0;
                    
                    inputs.forEach(input => {
                        if (input.value.trim() !== '') {
                            completed++;
                        }
                    });
                    
                    document.getElementById('completed').textContent = completed;
                }
                
                // 监听输入框变化以更新进度
                const inputs = document.querySelectorAll('.user-input');
                inputs.forEach(input => {
                    input.addEventListener('input', updateProgress);
                });
            });
